    vec3 SolveBarycentric(vec2 a, vec2 b, vec2 c, vec2 p) {
        vec2 v0 = b - a;
        vec2 v1 = c - a;
        vec2 v2 = p - a;
        f32 d00 = dot(v0, v0);
        f32 d10 = dot(v1, v0);
        f32 d11 = dot(v1, v1);
        f32 d20 = dot(v2, v0);
        f32 d21 = dot(v2, v1);
        f32 denom = d00 * d11 - d10 * d10;
        vec3 result; 
        result.y = (d20 * d11 - d10 * d21) / denom;
        result.z = (d00 * d21 - d20 * d10) / denom;
        result.x = 1.0f - result.y - result.z;
        return result;
    }
